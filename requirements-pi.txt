# =============================================================
#  Raspberry Pi — GPIO control dependencies (Pi ONLY)
# =============================================================
# Install on the Pi AFTER the main requirements.txt:
#
#     pip3 install -r requirements.txt
#     pip3 install -r requirements-pi.txt
#
# Why separate: these are Linux/Pi-only and must NOT be installed on the
# Windows dev machine.
#
# We use gpiozero (high-level API) with the lgpio backend. gpiozero
# auto-selects the lgpio pin factory on the Raspberry Pi 5's RP1 I/O
# controller, so the classic RPi.GPIO is neither needed nor compatible.
#
# NOTE: lgpio compiles from source when there's no prebuilt wheel for your
# Python version (e.g. Python 3.13). Install the build tools first:
#   sudo apt install -y swig python3-dev build-essential liblgpio-dev

gpiozero>=2.0
lgpio>=0.2.2.0
